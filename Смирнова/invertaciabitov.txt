MOV 0, R0
MOV @ARRAY, R1	; ПЕРЕМЕЩАЕМ МАССИВ В R1
MOV 0, R0	; ОБНУЛЯЕМ СУММУ
MOV 0A, R3	; УСТАНАВЛИВАЕМ СЧЕТЧИК 10

;ЦИКЛ
LOOP:
	MOV (R1), R2	; ПЕРЕМЕЩАЕМ ЭЛЕМЕНТ МАССИВА В R2
	AND 1,    R2	; ПРОВЕРКА МЛАДШЕГО БИТА
	JNZ NECHET	; ЕСЛИ НЕ 0, ЗНАЧИТ НЕЧЕТНОЕ, ПРЫГАЕМ
	MOV (R1), R2	; ЕСЛИ ЧЕТНОЕ, ПЕРЕМЕЩАЕМ ЭЛЕМ. СНОВА, Т.К AND УБИЛ ЧИСЛО
	XOR FF00, R2	; ИНВЕРТИРУЕМ СТАРШИЙ БАЙТ
	ADD R2,   R0	; ДОБАВЛЯЕМ К СУММЕ
	MOV R2,   (R1)+	; ЗАПОКОВЫВАЕМ ОБРАТНО В МАССИВ И ПЕРЕХОДИМ К СЛЕДУЮЩЕМУ ЭЛЕМ
	JMP NEXT	; ПРЫГАЕМ НА NEXT

	NECHET:
		MOV (R1), R2	; ПЕРЕМЕЩАЕМ ЭЛЕМ. СНОВА, Т.К AND УБИЛ ЧИСЛО
		XOR FF00, R2	; ИНВЕРТИРУЕМ
		MOV R2,   (R1)+	; ДОБАВЛЯЕМ В МАССИВ И ПЕРЕХОДИМ К СЛЕДУЮЩЕМУ ЭЛЕМ

	SUB 1, R3                  
	JNZ LOOP
	STOP

	NEXT:
		SUB 1, R3	; ОТНИМАЕМ 1 У СЧЕТЧИКА
		JNZ LOOP	; ЕСЛИ НЕ НОЛЬ, ВОЗВРАТ К ЦИКЛУ

MOV 0, R1
MOV 0, R2
MOV 0, R3

STOP

ARRAY:
	DATA FF01
	DATA FF02
	DATA FF03
	DATA FF04
	DATA FF05
	DATA FF06
	DATA FF07
	DATA FF08
	DATA FF09
	DATA FF0A
